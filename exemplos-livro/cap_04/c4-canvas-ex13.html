JavaScript

<script>
window.onload = function() {
var canvas = document.getElementById('quadro');
var ctx = canvas.getContext('2d'); 
	ctx.lineJoin = escolha;		
	ctx.lineCap = escolha;
	ctx.globalAlpha = escolha;
	
	ctx.lineWidth = 50;
	ctx.beginPath();
	ctx.moveTo(100,40);
	ctx.strokeStyle = '#f00';
	ctx.lineTo(100,130);
	ctx.lineTo(350,130);
	ctx.stroke();
}
</script>
