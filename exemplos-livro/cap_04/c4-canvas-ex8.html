Canvas - Exemplo 8: path (arcos).

Conteúdo alternativo para navegadores que não suportam canvas.

JavaScript

<script>
window.onload = function() {
var canvas = document.getElementById('quadro');
var ctx = canvas.getContext('2d'); 
	
	ctx.beginPath();
	ctx.moveTo(20,50);
	ctx.strokeStyle = '#f00';
	ctx.arcTo(100, 50, 100, 130, 80);
	ctx.stroke();
	
	ctx.beginPath();
	ctx.moveTo(100,130);
	ctx.strokeStyle = '#009';
	ctx.arcTo(200, 130, 200, 30, 100);
	ctx.stroke();
	
	ctx.beginPath();
	ctx.moveTo(200,30);
	ctx.strokeStyle = '#090';
	ctx.arcTo(200, 190, 380, 190, 180);
	ctx.stroke();
}
</script>

CSS

	canvas {border:1px solid #ccc;}

HTML

	<canvas id="quadro" width="400" height="200">
		<p>Conteúdo alternativo para navegadores que não suportam canvas.</p>
	</canvas>
