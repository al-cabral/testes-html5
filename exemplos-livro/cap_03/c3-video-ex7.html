JavaScript

<script type="text/javascript">
function id(id) {
	return document.getElementById(id)
}
function init() {	
	var container = id('fundo-controles');
	var ctr = id('video');
	var play = id('play');
	var pause = document.getElementById('pause');
	var vmais = document.getElementById('vmais');
	var vmenos = document.getElementById('vmenos');

	ctr.removeAttribute('controls');

	if (!Modernizr.audio) {
		container.style.display = 'none';	
		} else {	
	
	play.addEventListener('click', function(){
		ctr.play();
		ctr.volume = 1;
	}, false);

	pause.addEventListener('click', function(){
		ctr.pause();
	}, false);

	vmais.addEventListener('click', function(){
		ctr.play();
		ctr.volume += 0.1;
	
	}, false);

	vmenos.addEventListener('click', function(){
		ctr.play();
		ctr.volume -= 0.1;
	}, false);
	};
}
window.onload = init;
</script >
